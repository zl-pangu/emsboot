改变按钮状态

改变按钮状态，添加 data-loading-text="加载中..." 

<button type="button" id="myButton" data-loading-text="加载中..." class="btn btn-primary">加载状态</button>
<!-- loading button -->
<script>
  $('#myButton').on('click', function () {
var btn = $(this)
btn.button('loading');
setTimeout(function () {
  btn.button('reset');
}, 3000)
  })
</script>
